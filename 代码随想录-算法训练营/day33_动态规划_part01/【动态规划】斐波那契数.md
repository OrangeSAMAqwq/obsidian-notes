```cpp
class Solution {
public:
    int fib(int n) {
        if (n == 0)
            return 0;
        else if (n == 1)
            return 1;

        vector<int> cnt(n + 1, 0);
        cnt[0] = 0;
        cnt[1] = 1;
        for (int i = 2; i <= n; i++)
            cnt[i] = cnt[i - 1] + cnt[i - 2];
        return cnt[n];
    }
};
```
