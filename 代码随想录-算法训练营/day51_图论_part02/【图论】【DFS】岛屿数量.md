#include <cstdio>
#include <iostream>
#include <vector>
using namespace std;

const vector<int> Dx = {1, -1, 0, 0};
const vector<int> Dy = {0, 0, 1, -1};

void dfs(auto &grid, int i, int j, int N, int M){
    if (i < 0 || i >= N || j < 0 || j >= M)
        return;
    if (grid[i][j] == 0)
        return;

    grid[i][j] = 0;

    for (int k = 0; k < 4; k++){
        dfs(grid, i + Dx[k], j + Dy[k], N, M);
    }
}

int main(){
    int N, M;
    cin >> N >> M;

    vector<vector<bool>> grid(N, vector<bool>(M, false));
    for (int i = 0; i < N; i++){
        for (int j = 0; j < M; j++){
            bool cnt;
            cin >> cnt;
            grid[i][j] = cnt;
        }
    }

    int ans = 0;
    for (int i = 0; i < N; i++){
        for (int j = 0; j < M; j++){
            if (grid[i][j]){
                ans++;
                dfs(grid, i, j, N, M);
            }
        }
    }

    cout << ans << endl;
    return 0;
}
