#include <cstdio>
#include <iostream>
#include <vector>
#include <queue>
using namespace std;

const vector<int> Dx = {1, -1, 0, 0};
const vector<int> Dy = {0, 0, 1, -1};

int main(){
    int N, M;
    cin >> N >> M;

    vector<vector<bool>> grid(N, vector<bool>(M, false));
    for (int i = 0; i < N; i++){
        for (int j = 0; j < M; j++){
            bool cnt;
            cin >> cnt;
            grid[i][j] = cnt;
        }
    }

    int ans = 0;
    for (int i = 0; i < N; i++){
        for (int j = 0; j < M; j++){
            if (grid[i][j]){
                ans++;

                queue<vector<int>> q;
                q.push({i, j});
                grid[i][j] = false;

                while (!q.empty()){
                    vector<int> cur = q.front();
                    q.pop();

                    for (int d = 0; d < 4; d++){
                        int xx = cur[0] + Dx[d];
                        int yy = cur[1] + Dy[d];

                        if (xx >= 0 && xx < N &&
                            yy >= 0 && yy < M &&
                            grid[xx][yy]){
                            grid[xx][yy] = false;
                            q.push({xx, yy});
                        }
                    }
                }
            }
        }
    }

    cout << ans << endl;
    return 0;
}
