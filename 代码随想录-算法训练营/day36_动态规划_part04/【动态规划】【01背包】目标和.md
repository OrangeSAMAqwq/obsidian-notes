class Solution {
public:
    int findTargetSumWays(vector<int>& nums, int target) {
        int sumVal = 0;
        for (int i = 0; i < nums.size(); i++){
            sumVal += nums[i];
        }
        if (target > sumVal || target < -sumVal)
            return 0;

        vector<int> dp(sumVal * 2 + 2, 0);
        dp[0] = 1;

        for (int i = 0; i < nums.size(); i++){
            for (int j = sumVal * 2 + 1; j >= 2 * nums[i]; j--){
                if (dp[j - 2 * nums[i]] != 0)
                    dp[j] += dp[j - 2 * nums[i]];
            }
        }
        return dp[target + sumVal];
    }
};
